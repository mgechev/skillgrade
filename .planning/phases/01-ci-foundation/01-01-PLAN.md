**2. Create composite action**

Create `.github/actions/setup-node/action.yml` with this exact content:

```yaml
name: 'Setup Node.js'
description: 'Install Node.js from .node-version and run npm ci'
runs:
  using: 'composite'
  steps:
    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version-file: '.node-version'
        cache: 'npm'
    - name: Install dependencies
      run: npm ci
      shell: bash
```

Key constraints:
- `shell: bash` is REQUIRED on the `run:` step (composite actions do not inherit the workflow shell)
- Does NOT include `actions/checkout` (each job checks out separately per locked decision)
- Uses `actions/setup-node@v4` (locked decision, not v6)
- Uses `cache: 'npm'` for CI-02 requirement (npm package caching)
- Uses `node-version-file: '.node-version'` to read Node version from single source of truth

Create `.github/workflows/ci.yml` with the following exact structure. Every detail below is from locked decisions in CONTEXT.md:

```yaml
name: CI

on:
  pull_request:
  push:
    branches: [main]
  workflow_dispatch:

permissions:
  contents: read

concurrency:
  group: ${{ github.workflow }}-${{ github.event.pull_request.number || github.ref }}
  cancel-in-progress: true

env:
  NODE_OPTIONS: '--max-old-space-size=4096'

jobs:
  typecheck:
    name: Typecheck
    runs-on: ubuntu-24.04-arm
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: ./.github/actions/setup-node
      - run: npm run typecheck

  build:
    name: Build
    runs-on: ubuntu-24.04-arm
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: ./.github/actions/setup-node
      - run: npm run build

  test-bootstrap:
    name: Test (bootstrap)
    runs-on: ubuntu-24.04-arm
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: ./.github/actions/setup-node
      - run: npm run test:bootstrap

  test-analytics:
    name: Test (analytics)
    runs-on: ubuntu-24.04-arm
    timeout-minutes: 20
    steps:
      - uses: actions/checkout@v4
      - uses: ./.github/actions/setup-node
      - run: npm run test:analytics
```
